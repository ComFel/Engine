524/* Note that memset() is a byte assignment and this is a 32-bit assignment, so they're not directly equivalent. */
525SDL_FORCE_INLINE void SDL_memset4(void *dst, Uint32 val, size_t dwords)
526{
527#if defined(__GNUC__) && defined(__i386__)
528 int u0, u1, u2;
529 __asm__ __volatile__ (
530 "cld \n\t"
531 "rep ; stosl \n\t"
532 : "=&D" (u0), "=&a" (u1), "=&c" (u2)
533 : "0" (dst), "1" (val), "2" (SDL_static_cast(Uint32, dwords))
534 : "memory"
535 );
536#else
537 size_t _n = (dwords + 3) / 4;
538 Uint32 *_p = SDL_static_cast(Uint32 *, dst);
539 Uint32 _val = (val);
540 if (dwords == 0) {
541 return;
542 }
543 switch (dwords % 4) {
544 case 0: do { *_p++ = _val; SDL_FALLTHROUGH;
545 case 3: *_p++ = _val; SDL_FALLTHROUGH;
546 case 2: *_p++ = _val; SDL_FALLTHROUGH;
547 case 1: *_p++ = _val;
548 } while ( --_n );
549 }
550#endif
551}
